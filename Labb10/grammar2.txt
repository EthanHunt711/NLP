# Grammar
S -> S Punct
S -> NP VP
NP -> Pronoun | Det | Noun | Adj Noun
NP -> NP PP
VP -> VP NP
VP -> Verb PP PP
VP -> Aux Verb
VP -> NP PP
PP -> Prep NP
# Lexicon
Noun -> 'BBC' | 'decade' | 'CNN' | 'Style' | 'change' | 'pattern' | 'seasons' | 'experience'
Verb -> 'worked' | 'caused' | 'spoke'
Pronoun -> 'He' | 'She'
Det -> 'the' | 'a'
Prep -> 'for' | 'in' | 'of' | 'to' | 'about'
Punct -> '.'
Adj -> 'Global' | 'rainy'
Aux -> 'has'
